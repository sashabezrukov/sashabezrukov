public class MaximumNumber {
    public static void main(String[] args) {
        int[] numbers = {5, 10, 2, 8, 3};
        int max = numbers[0];

        for (int i = 1; i < numbers.length; i++) {
            if (numbers[i] > max) {
                max = numbers[i];
            }
        }

        System.out.println("The maximum number is: " + max);
    }
}
